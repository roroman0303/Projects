// создаватель двумерного дин. массива 
float twoDimMatrixCreator (int row_count, int col_count) 
{
    float** element = new float*[row_count]; 
    for (int i = 0; i < row_count; i++) 
    {
        element[i] = new float[col_count];
    }
    return element;
}
