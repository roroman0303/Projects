//Алгоритма поиска максимального и 
//минимального элементов одномерного массива
//и замена первого элемента  их суммой.
#include <stdio.h>
#include <conio.h>
using namespace std;

int main()
{
    float x[10];
    float max, min;

    for (int i=0; i < 10 ; i++)
    {
        cout<< "x["<<(i+1)<<"]= ";
        cin>>x[i];
    }

//Поиск максимального элемента и его номера
    max=x[0];
    min=x[0];
    for(int i=1; i < 10 ; i++)
    {
        if (x[i]>max)
        {
            max = x[i];
        }
        if (x[i]<min)
        {
            min=x[i];
        }
    }
//замена суммой первого элемента массива
    x[0] = max+min;
//Вывод элементов массива
    cout<<"max = "<<max;
    cout<<"min = "<<min;

    for(int i=0; i < 10 ; i++)
    {
        cout<<"x["<<(i+1)<<"] = "<<x[i];
    }
    return 0;
}
