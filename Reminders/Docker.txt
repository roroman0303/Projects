# начало работы
git clone https://... - для клонирования образа с гита
cd ./папка - для перехода в папку
docker build -t name_of_image . - для сборки образа с названием из директории .
docker rm - удаление контейнера
docker rmi - удаление образа

# запуск образа
docker run name_of_image - запуск образа
	-d - для запуска в автономном режиме (т.е. продолжит работать даже при выходе из терминала)
	-p - для публикации портов контейнера на хосте
	--name - для присвоения имени контейнеру
	-w - для указания рабочего каталога
	-e - устанавливает переменные окружения в формате пар КЛЮЧ=ЗНАЧЕНИЕ
	--env-file - устанавливает переменные окружения из файла
	--ip - назначает docker-контейнеру IP-адрес
	--rm - автоматически удалить контейнер после завершения его работы 
docker stop name_of_image - остановить контейнер
docker stop $(docker ls -a -q). - остановить все контейнеры
docker kill name_of_image - принудительно остановить контейнер

# другое
docker ls - посмотреть запущенные контейнеры
docker logs name_of_image - смотреть логи контейнера
	-follow - логи работающего контейнера
